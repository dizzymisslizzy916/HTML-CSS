// This is a program that will determine if a number entered by the user is perfect or not. That is determined by finding all the divisors of the number (except itself) and comparing the sum of divisors to the original number.

import java.util.*;
import java.text.*;

class PerfectLC{
public static void main (String [] args){
Scanner keyboard = new Scanner(System.in);

int num = 0; // Declaring all the needed variables to do the calculations
int remainder;
int count = 0;
int sum = 0;
int [] divisors;
int count2 = 0;
boolean flag = true;

System.out.println("Hello! This is a program that will tell you if a number is perfect or not.");
while(flag){ // This loop will keep the program running until the user enters a whole number
System.out.println("Enter a whole number: ");
try{
num = keyboard.nextInt();
if(num > 0){ // Catches negative numbers
throw new Exception("Please enter a positive number.");
}
flag = false;
} catch (InputMismatchException e) { // Catches invalid input
System.out.println("That's not a whole number.");
keyboard.next();
} catch (Exception e) {
System.out.println(e.getMessage());
}
}

for(int i = 1; i remainder = num % i; // This finds if the number i can be divided evenly into the user's number
if (remainder == 0){ // If remainder equals 0, it's a divisor
if(count == 0){
System.out.print("Divisors:");
}
System.out.print(" " + i); // Here the divisors are printed
count++; // Count goes up each time a divisor is found, giving us the total number of divisors
um = sum + i; // Here the divisors are added together
}
}

divisors = new int [count]; // Now we can assign the array size to the total number of divisors
for(int i = 1; i < num; i++){ // This loops accomplishes the same task as the previous for loop
remainder = num % i;
if (remainder == 0){
divisors[count2] = i; // If it's a divisor, it's stored in the array here
count2++; // Count2 gives the index of the stored divisor
}
}

System.out.println();
System.out.println("Sum: " + sum);
if(sum == num){ // If the sum equals the inputted number, it's perfect
System.out.println("Yes, that is a perfect number.");
} else {
System.out.println("No, that isn't a perfect number.");
}
System.out.println("Goodbye.");
}
}
